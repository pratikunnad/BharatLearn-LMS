1. Hello World
#include <iostream>
using namespace std;

int main() {
    cout << "Hello World";
    return 0;
}
2. Add Two Numbers
#include <iostream>
using namespace std;

int main() {
    int a, b;
    cin >> a >> b;
    cout << "Sum = " << a + b;
    return 0;
}
3. Check Even or Odd
#include <iostream>
using namespace std;

int main() {
    int num;
    cin >> num;

    if (num % 2 == 0)
        cout << "Even";
    else
        cout << "Odd";

    return 0;
}
4. Factorial of a Number
#include <iostream>
using namespace std;

int main() {
    int num;
    long long fact = 1;
    cin >> num;

    for (int i = 1; i <= num; i++)
        fact *= i;

    cout << "Factorial = " << fact;
    return 0;
}
5. Check Prime Number
#include <iostream>
using namespace std;

int main() {
    int num;
    cin >> num;
    bool prime = true;

    if (num <= 1) prime = false;

    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0) {
            prime = false;
            break;
        }
    }

    cout << (prime ? "Prime" : "Not Prime");
    return 0;
}
