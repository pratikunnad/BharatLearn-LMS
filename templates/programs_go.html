2. Add Two Numbers

package main
import "fmt"

func main() {
    var a, b int
    fmt.Print("Enter two numbers: ")
    fmt.Scan(&a, &b)
    fmt.Println("Sum =", a+b)
}

3. Check Even or Odd

package main
import "fmt"

func main() {
    var n int
    fmt.Print("Enter a number: ")
    fmt.Scan(&n)

    if n%2 == 0 {
        fmt.Println("Even")
    } else {
        fmt.Println("Odd")
    }
}

4. Factorial of a Number

package main
import "fmt"

func main() {
    var n, fact int = 1, 1
    fmt.Print("Enter number: ")
    fmt.Scan(&n)

    for i := 1; i <= n; i++ {
        fact *= i
    }
    fmt.Println("Factorial =", fact)
}

5. Reverse a Number

package main
import "fmt"

func main() {
    var n, rev int
    fmt.Print("Enter number: ")
    fmt.Scan(&n)

    for n != 0 {
        rev = rev*10 + n%10
        n /= 10
    }
    fmt.Println("Reversed =", rev)
}
