    var num int

    }
}
4. Factorial of a Number
package main

import "fmt"

func main() {
    var num int
    var fact int = 1
    fmt.Scan(&num)

    for i := 1; i <= num; i++ {
        fact *= i
    }

    fmt.Println("Factorial =", fact)
}
5. Check Prime Number
package main

import "fmt"

func main() {
    var num int
    fmt.Scan(&num)
    isPrime := true

    if num <= 1 {
        isPrime = false
    }

    for i := 2; i <= num/2; i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }

    if isPrime {
        fmt.Println("Prime")
    } else {
        fmt.Println("Not Prime")
    }
}
