3. Check Even or Odd
#include <stdio.h>

int main() {
    int num;
    scanf("%d", &num);

    if (num % 2 == 0)
        printf("Even");
    else
        printf("Odd");

    return 0;
}
4. Factorial of a Number
#include <stdio.h>

int main() {
    int num;
    long fact = 1;
    scanf("%d", &num);

    for (int i = 1; i <= num; i++)
        fact *= i;

    printf("Factorial = %ld", fact);
    return 0;
}
5. Check Prime Number
#include <stdio.h>

int main() {
    int num, flag = 0;
    scanf("%d", &num);

    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0) {
            flag = 1;
            break;
        }
    }

    if (num <= 1 || flag)
        printf("Not Prime");
    else
        printf("Prime");

    return 0;
}
